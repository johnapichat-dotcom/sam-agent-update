while ($true) {
    
    $CurrentTime = Get-Date
    $HasSyncedToday = $false

    # --- Phase 1: Catch-Up & State Awareness ---
    if (Test-Path $StateFile) {
        $LastSyncTime = (Get-Item $StateFile).LastWriteTime
        if ($LastSyncTime.Date -eq $CurrentTime.Date) {

        if ($Silent) {
            if ($CurrentTime.Hour -ge 8 -and $CurrentTime.Hour -lt 14) {
                # Prime Time: เกลี่ยโหลดตามเวลาที่เหลือจนถึง 14:00 น.
                $EndTime = Get-Date -Hour 14 -Minute 0 -Second 0
                $RemainingSeconds = ($EndTime - $CurrentTime).TotalSeconds
                $RandomWait = Get-Random -Minimum 10 -Maximum [int]$RemainingSeconds
                "$(Get-Date) - [Queue] Prime Window (8-14). Delaying $RandomWait seconds." | Out-File "$WorkDir\SAM_Log.txt" -Append
                Start-Sleep -Seconds $RandomWait
            }
            elseif ($CurrentTime.Hour -ge 14) {
                # Latecomer: เก็บตกเครื่องเปิดสาย
                $RandomWait = Get-Random -Minimum 60 -Maximum 300
                "$(Get-Date) - [Queue] Latecomer. Quick sync in $RandomWait seconds." | Out-File "$WorkDir\SAM_Log.txt" -Append
                Start-Sleep -Seconds $RandomWait
            }
            else {
                # Early Bird: เครื่องเปิดเช้าตรู่ก่อน 8 โมง
                $RandomWait = Get-Random -Minimum 300 -Maximum 600
                "$(Get-Date) - [Queue] Early bird. Sync in $RandomWait seconds." | Out-File "$WorkDir\SAM_Log.txt" -Append
                Start-Sleep -Seconds $RandomWait
            }
        }

        # --- Phase 3: Execution & Failover ---

        $MaxWaitLoops = 24
        $CurrentLoop = 0
        $InternetReady = $false

        while ($CurrentLoop -lt $MaxWaitLoops -and -not $InternetReady) {
            try {
                $NetCheck = Invoke-WebRequest -Uri "http://www.msftconnecttest.com/connecttest.txt" -UseBasicParsing -TimeoutSec 10 -ErrorAction Stop
                if ($NetCheck.Content -match "Microsoft Connect Test") { $InternetReady = $true } else { throw "Captive Portal Detected" }
            } catch {
                $CurrentLoop++
                if (-not $Silent) { Write-Host "[Warning] Network offline. Retrying in 5 minutes... ($CurrentLoop/$MaxWaitLoops)" -ForegroundColor Yellow }
                Start-Sleep -Seconds 300 
            }
        }

        if (-not $InternetReady) { 
            "$(Get-Date) - [Offline] No network after 2 hours. Hibernating till next cycle." | Out-File "$WorkDir\SAM_Log.txt" -Append
        } else {

                            $RunMode = if ($Silent) { "-Silent" } else { "" }

            # 3.2 Security & AV

            $HTML += "<h2>Security & Antivirus Status</h2><table><tr><th>Installed Antivirus</th><th>Version</th></tr>"
            $SoftwarePaths = @("HKLM:\Software\Microsoft\Windows\CurrentVersion\Uninstall\*", "HKLM:\Software\Wow6432Node\Microsoft\Windows\CurrentVersion\Uninstall\*")
            $AllSoftware = Get-ItemProperty $SoftwarePaths -ErrorAction SilentlyContinue | Where-Object DisplayName -ne $null
            try {
                $AVs = Get-CimInstance -Namespace "root\SecurityCenter2" -ClassName AntivirusProduct -ErrorAction Stop
                if ($AVs) {
                    foreach ($AV in $AVs) {
                        $AVName = $AV.displayName
                        $AVVer = "Unknown"
                        if ($AVName -match "Windows Defender" -or $AVName -match "Microsoft Defender") {
                            $MpStatus = Get-MpComputerStatus -ErrorAction SilentlyContinue
                            if ($MpStatus -and $MpStatus.AMProductVersion) { $AVVer = $MpStatus.AMProductVersion } else { $AVVer = "Built-in OS Version" }
                        } else {
                            $SearchName = $AVName -replace " Antivirus","" -replace " Security","" -replace " Endpoint",""
                            $MatchedApp = $AllSoftware | Where-Object { $_.DisplayName -match [regex]::Escape($SearchName) } | Select-Object -First 1
                            if ($MatchedApp -and $MatchedApp.DisplayVersion) { $AVVer = $MatchedApp.DisplayVersion }
                        }
                        $HTML += "<tr><td style='font-weight:bold; color:green;'>$AVName</td><td style='font-weight:bold;'>$AVVer</td></tr>"
                        $CloudAV += "$AVName (v.$AVVer)"
                    }
                } else { $HTML += "<tr><td colspan='2' style='color:red; font-weight:bold;'>No Antivirus Detected</td></tr>" }
            } catch { $HTML += "<tr><td colspan='2'>Windows Defender or WMI Error</td></tr>"; $CloudAV += "WMI Error / Defender" }
            $HTML += "</table>"
            $CloudAV1 = if ($CloudAV.Count -ge 1) { $CloudAV[0] } else { "Not Detected" }
            $CloudAV2 = if ($CloudAV.Count -ge 2) { $CloudAV[1..($CloudAV.Count-1)] -join " | " } else { "-" }

            # 3.3 Health Metrics
            $Stability = (Get-CimInstance Win32_ReliabilityStabilityMetrics -ErrorAction SilentlyContinue | Sort-Object timegenerated -Descending | Select-Object -First 1).SystemStabilityIndex
            $HealthScore = if ($null -ne $Stability) { "$([math]::Round($Stability, 1))/10" } else { "N/A" }
            $BootTime = (Get-CimInstance Win32_OperatingSystem).LastBootUpTime
            $UptimeDays = ((Get-Date) - $BootTime).Days
            $UptimeStr = if ($UptimeDays -ge 14) { "$UptimeDays Days (Warning)" } else { "$UptimeDays Days" }
            
            $Pending1 = Test-Path "HKLM:\SOFTWARE\Microsoft\Windows\CurrentVersion\Component Based Servicing\RebootPending"
            $Pending2 = Test-Path "HKLM:\SOFTWARE\Microsoft\Windows\CurrentVersion\WindowsUpdate\Auto Update\RebootRequired"
            $Pending3 = $null -ne (Get-ItemProperty -Path "HKLM:\SYSTEM\CurrentControlSet\Control\Session Manager" -Name "PendingFileRenameOperations" -ErrorAction SilentlyContinue)
            $NeedsReboot = if ($Pending1 -or $Pending2 -or $Pending3) { "Yes" } else { "No" }
            $HTML += "<h2>System Health & Stability Metrics</h2><table><tr><th>Stability Score</th><th>System Uptime</th><th>Pending Reboot</th></tr><tr><td>$HealthScore</td><td>$UptimeStr</td><td>$NeedsReboot</td></tr></table>"

            # 3.4 Local Users
            $CloudActiveUsers = @()
            $HTML += "<h2>Local User Accounts</h2><table><tr><th>Account Name</th><th>Role</th><th>Status</th></tr>"

                $Role = if ($AdminGrp -match $_.Name) { "Administrator" } else { "Standard User" }

                if ($_.Enabled -eq $true) { $CloudActiveUsers += "$($_.Name) ($Role)" }
            }
            $FinalCloudUsers = $CloudActiveUsers -join " | "
            $HTML += "</table>"

            # 3.5 Software & Printers
            $CloudSoftwareList = @()
            $HTML += "<h2>Installed Applications</h2><table><tr><th>Application Name</th><th>Version</th></tr>"
            $AllSoftware | Sort-Object DisplayName -Unique | ForEach-Object {
                $HTML += "<tr><td>$($_.DisplayName)</td><td>$($_.DisplayVersion)</td></tr>"
                $CloudSoftwareList += "$($_.DisplayName)"
            }
            $FinalCloudSoftware = $CloudSoftwareList -join " | "
            $HTML += "</table>"

            $CloudPrinterList = @()
            $HTML += "<h2>Printers & OPOS</h2><table><tr><th>Name</th></tr>"
            Get-Printer | Sort-Object Name | ForEach-Object { $HTML += "<tr><td>$($_.Name)</td></tr>"; $CloudPrinterList += "$($_.Name)" }
            $OPOS_Paths = @("HKLM:\SOFTWARE\OLEforRetail\ServiceOPOS\POSPrinter", "HKLM:\SOFTWARE\WOW6432Node\OLEforRetail\ServiceOPOS\POSPrinter", "HKLM:\SOFTWARE\Epson\EPSON OPOS\POSPrinter", "HKLM:\SOFTWARE\WOW6432Node\Epson\EPSON OPOS\POSPrinter", "HKLM:\SOFTWARE\Star Micronics\OPOS\POSPrinter", "HKLM:\SOFTWARE\WOW6432Node\Star Micronics\OPOS\POSPrinter")
            foreach ($Path in $OPOS_Paths) { if (Test-Path $Path) { foreach ($Dev in Get-ChildItem -Path $Path -ErrorAction SilentlyContinue) { $HTML += "<tr><td style='color:blue'>[OPOS] $($Dev.PSChildName)</td></tr>"; $CloudPrinterList += "[OPOS] $($Dev.PSChildName)" } } }
            $FinalCloudPrinters = $CloudPrinterList -join " | "
            $HTML += "</table>"

            # 3.6 Disk Health
            $CloudDiskList = @()
            $HTML += "<h2>Disk Health</h2><table><tr><th>Model</th><th>Status</th><th>Capacity (GB)</th></tr>"
            try {
                $Disks = Get-PhysicalDisk -ErrorAction Stop
                $DiskCount = 1
                foreach ($Disk in $Disks) {
                    $SizeGB = [math]::Round($Disk.Size / 1GB, 2)
                    $HStatus = $Disk.HealthStatus
                    $HTML += "<tr><td>$($Disk.Model)</td><td>$HStatus</td><td>$SizeGB</td></tr>"
                    $CloudDiskList += "[Disk $DiskCount] Model: $($Disk.Model) | Status: $HStatus | Size: $SizeGB GB"
                    $DiskCount++
                }
            } catch { $CloudDiskList += "WMI Error" }
            $HTML += "</table>"
            $CloudDiskHealth = $CloudDiskList -join "`n"

            # 3.7 Network
            $Net = Get-WmiObject Win32_NetworkAdapterConfiguration | Where-Object { $_.IPAddress -ne $null -and $_.IPAddress -notmatch ":" } | Select-Object -First 1

            if ($DeepScan) {
                # (Deep Scan Functions - Keep original structure if triggered manually)
                $HTML += "<h2>Deep Scan Mode Executed</h2>"
            }

            # 4. DATA SYNCHRONIZATION ENGINE (DELTA UPDATE)
            $CurrentData = [ordered]@{
                BU             = $BU
                BranchID       = $BranchID
                ComputerName   = $env:COMPUTERNAME
                ComputerDesc   = $CompDesc
                Manufacturer   = $CompSys.Manufacturer
                Model          = $CompSys.Model
                SerialNumber   = $BIOS.SerialNumber
                OSVersion      = $CombinedOS
                CPU            = $CPU.Name
                RAM            = "$RAM_Total GB"
                IPAddress      = $Net.IPAddress[0]
                MACAddress     = $Net.MACAddress
                Antivirus1     = $CloudAV1
                Antivirus2     = $CloudAV2
                DiskHealth     = $CloudDiskHealth
                StabilityScore = $HealthScore
                Uptime         = $UptimeStr
                PendingReboot  = $NeedsReboot
                TeamViewerID   = $TV_ID
                AnyDeskID      = $AD_ID
                ActiveUsers    = $FinalCloudUsers
                AllPrinters    = $FinalCloudPrinters
                AllSoftware    = $FinalCloudSoftware
                AgentVersion   = $AgentVersion
            }

            $SyncType = "FullSync"
            $FinalPayload = @{}

            if (Test-Path $StateFile) {
                try {
                    $LastState = Get-Content $StateFile -Raw | ConvertFrom-Json
                    $ChangesFound = $false
                    
                    $FinalPayload["SerialNumber"] = $CurrentData["SerialNumber"]
                    $FinalPayload["ComputerName"] = $CurrentData["ComputerName"]
                    $FinalPayload["BranchID"]     = $CurrentData["BranchID"]
                    $FinalPayload["BU"]           = $CurrentData["BU"]

                    foreach ($Key in $CurrentData.Keys) {
                        $CurrentVal = [string]$CurrentData[$Key]
                        $LastVal    = [string]$LastState.$Key
                        if ($CurrentVal -cne $LastVal) {
                            $FinalPayload[$Key] = $CurrentVal
                            $ChangesFound = $true
                        }
                    }
                    $SyncType = if ($ChangesFound) { "DeltaUpdate" } else { "Heartbeat" }
                    $FinalPayload["Type"] = $SyncType
                } catch { $SyncType = "FullSync" }
            }

            if ($SyncType -eq "FullSync") {
                $FinalPayload = $CurrentData
                $FinalPayload["Type"] = "FullSync"
            }

            # 5. CLOUD DATA TRANSMISSION & C2 (WITH RANDOM RETRY & STOPWATCH)
            $MaxRetries = 6        
            $Attempt = 0
            $IsSuccess = $false

            $JsonPayload = $FinalPayload | ConvertTo-Json -Depth 5
            $Utf8Bytes = [System.Text.Encoding]::UTF8.GetBytes($JsonPayload)

            while ($Attempt -lt $MaxRetries -and -not $IsSuccess) {
                $Attempt++
                try {
                    $SyncTimer = [System.Diagnostics.Stopwatch]::StartNew()
                    $Response = Invoke-RestMethod -Uri $GoogleScriptURL -Method Post -Body $Utf8Bytes -ContentType "application/json; charset=utf-8" -TimeoutSec 30
                    $SyncTimer.Stop()
                    $IsSuccess = $true
                    
                    $ElapsedSec = [math]::Round($SyncTimer.Elapsed.TotalSeconds, 2)
                    $PayloadSizeKB = [math]::Round($Utf8Bytes.Length / 1KB, 2)
                    "$(Get-Date) - [Cloud Sync] Attempt $Attempt: Success. Size: $PayloadSizeKB KB | Time: $ElapsedSec s." | Out-File "$WorkDir\SAM_Log.txt" -Append

                    if ($Response -match "Success" -or $Response -match "Ack") {
                        $CurrentData | ConvertTo-Json -Depth 5 | Out-File $StateFile -Encoding UTF8
                        
                        # --- C2 COMMAND PROCESSING & SELF-HEALING ---
                        if ($Response -match "CMD:(.+)") {
                            $RemoteCmd = $matches[1].Trim()
                            if ($RemoteCmd -ne "NONE" -and $RemoteCmd -ne "") {
                                "$(Get-Date) - C2 Command Received: $RemoteCmd" | Out-File "$WorkDir\SAM_Log.txt" -Append
                                
                                $CommandList = $RemoteCmd -split "&&"
                                foreach ($SingleCmd in $CommandList) {
                                    $ParsedCmd = $SingleCmd.Trim()
                                    switch -Regex ($ParsedCmd) {
                                        "^SHUTDOWN$"   { Start-Process "shutdown.exe" -ArgumentList "/s /t 60" -WindowStyle Hidden }
                                        "^RESTART$"    { Start-Process "shutdown.exe" -ArgumentList "/r /t 60" -WindowStyle Hidden }
                                        "^CLEAR_TEMP$" { Remove-Item -Path "$env:TEMP\*" -Recurse -Force -ErrorAction SilentlyContinue }
                                        
                                        # [NEW] SELF-HEALING: ลบความจำเพื่อซ่อมแซมตัวเองในรอบถัดไป
                                        "^FORCE_SYNC$" { 
                                            Remove-Item -Path $StateFile -Force -ErrorAction SilentlyContinue
                                            "$(Get-Date) - [Self-Healing] State cleared. Forcing Full Sync next cycle." | Out-File "$WorkDir\SAM_Log.txt" -Append
                                        }
                                        
                                        # TASK PAYLOAD MODULE
                                        "^RUN_TASK:(.+)$" {
                                            $TaskData = $matches[1].Trim()
                                            $TaskParts = $TaskData -split '\|'
                                            $TaskFileName = $TaskParts[0].Trim()
                                            $TaskArgs = if ($TaskParts.Count -gt 1) { $TaskParts[1..($TaskParts.Count-1)] } else { @() }
                                            
                                            $TaskUrl = "$TaskBaseUrl$TaskFileName"
                                            $LocalTaskPath = "$env:TEMP\$TaskFileName"
                                            
                                            try {
                                                Invoke-WebRequest -Uri $TaskUrl -OutFile $LocalTaskPath -UseBasicParsing -ErrorAction Stop
                                                if (Test-Path $LocalTaskPath) {
                                                    $ArgString = ($TaskArgs | ForEach-Object { "`"$_`"" }) -join " "
                                                    $ProcessArgs = "-WindowStyle Hidden -ExecutionPolicy Bypass -Command `"& { try { & '$LocalTaskPath' $ArgString } finally { Start-Sleep -Seconds 2; Remove-Item -Path '$LocalTaskPath' -Force -ErrorAction SilentlyContinue } }`""
                                                    Start-Process powershell.exe -ArgumentList $ProcessArgs -WindowStyle Hidden
                                                }
                                            } catch { "$(Get-Date) - Task Failed: $TaskFileName" | Out-File "$WorkDir\SAM_Log.txt" -Append }
                                        }
                                    }
                                }
                            }
                        }
                    }
                } catch { 
                    if ($null -ne $SyncTimer) { $SyncTimer.Stop() }
                    $ElapsedFail = if ($null -ne $SyncTimer) { [math]::Round($SyncTimer.Elapsed.TotalSeconds, 2) } else { 0 }
                    
                    # Random Retry (5-15 mins) ป้องกันข้อมูลชนกันตอนเน็ตกลับมาพร้อมกัน
                    $RandomRetry = Get-Random -Minimum 300 -Maximum 900
                    "$(Get-Date) - [Sync Error] Attempt $Attempt Failed ($ElapsedFail s). Retrying in $RandomRetry seconds." | Out-File "$WorkDir\SAM_Log.txt" -Append
                    
                    if ($Attempt -lt $MaxRetries) { Start-Sleep -Seconds $RandomRetry } 
                }
            }

            $HTML += "</body></html>"
            [System.IO.File]::WriteAllText($ReportPath, $HTML, [System.Text.Encoding]::UTF8)
            if (-not $Silent) { Invoke-Item $ReportPath }
        }
    }

    # --- Phase 4: Deep Sleep (จำศีลข้ามวันสำหรับเครื่อง 24/7) ---
    if ($Silent) {
        $Now = Get-Date
        $Next8AM = Get-Date -Hour 8 -Minute 0 -Second 0
        
        # ถ้าตอนนี้เลย 8 โมงเช้าไปแล้ว ให้นับ 8 โมงเช้าของ "วันพรุ่งนี้"
        if ($Now.Hour -ge 8) { $Next8AM = $Next8AM.AddDays(1) }
        
        $SleepSeconds = [int]($Next8AM - $Now).TotalSeconds
        if ($SleepSeconds -lt 0) { $SleepSeconds = 60 } # Fallback ป้องกันค่าติดลบ
        
        "$(Get-Date) - [Immortal Loop] Cycle complete. Hibernating for $SleepSeconds seconds until 08:00 AM." | Out-File "$WorkDir\SAM_Log.txt" -Append
        Start-Sleep -Seconds $SleepSeconds
    } else {
        # ถ้ารันแบบ Manual (ไม่ Silent) ให้จบการทำงานทันที
        "$(Get-Date) - [Manual Mode] Execution complete. Exiting." | Out-File "$WorkDir\SAM_Log.txt" -Append
        break
    }
}
