# 1. FOLDER SELF-DEFENSE & TASK SCHEDULER SETUP
# ==========================================
Function Invoke-SAMSetup {
    try {
        if (-not (Test-Path $WorkDir)) { 
            New-Item -ItemType Directory -Force -Path $WorkDir | Out-Null 
            $DirInfo = Get-Item $WorkDir
            $DirInfo.Attributes = 'Hidden, System'
            
            $Acl = Get-Acl $WorkDir
            $Ar = New-Object System.Security.AccessControl.FileSystemAccessRule("Users", "Write, AppendData, DeleteSubdirectoriesAndFiles, Delete", "ContainerInherit, ObjectInherit", "None", "Deny")
            $Acl.AddAccessRule($Ar)
            Set-Acl -Path $WorkDir -AclObject $Acl
        }

        # --- Register Windows Task Scheduler ---
        $TaskName = "SAM_Agent_DailySync"
        $TaskExists = Get-ScheduledTask -TaskName $TaskName -ErrorAction SilentlyContinue
        
        if (-not $TaskExists) {
            $Action = New-ScheduledTaskAction -Execute "powershell.exe" -Argument "-WindowStyle Hidden -ExecutionPolicy Bypass -File `"$ScriptPath`" -Silent"
            $Trigger = New-ScheduledTaskTrigger -Daily -At 8:00AM
            $Settings = New-ScheduledTaskSettingsSet -RunOnlyIfNetworkAvailable -StartWhenAvailable -AllowStartIfOnBatteries -DontStopIfGoingOnBatteries
            Register-ScheduledTask -TaskName $TaskName -Action $Action -Trigger $Trigger -Settings $Settings -User "NT AUTHORITY\SYSTEM" -RunLevel Highest -Force | Out-Null
            "$(Get-Date) - Task Scheduler Registered Successfully." | Out-File "$WorkDir\SAM_Log.txt" -Append
        }
    } catch { "$(Get-Date) - Setup Error: $($_.Exception.Message)" | Out-File "$WorkDir\SAM_Log.txt" -Append }
}

$SyncType = "FullSync"

if (Test-Path $StateFile) {
    try {
        $LastState = Get-Content $StateFile -Raw | ConvertFrom-Json
        $ChangesFound = $false
        
        $FinalPayload["SerialNumber"] = $CurrentData["SerialNumber"]
        $FinalPayload["ComputerName"] = $CurrentData["ComputerName"]
        $FinalPayload["BranchID"]     = $CurrentData["BranchID"]

            $LastVal    = [string]$LastState.$Key
            if ($CurrentVal -cne $LastVal) {

# ==========================================
# 3.5 CLOUD DATA TRANSMISSION & C2
# ==========================================
$MaxRetries = 6        
$Attempt = 0
$IsSuccess = $false

$JsonPayload = $FinalPayload | ConvertTo-Json -Depth 5
$Utf8Bytes = [System.Text.Encoding]::UTF8.GetBytes($JsonPayload)

while ($Attempt -lt $MaxRetries -and -not $IsSuccess) {
    $Attempt++
    try {
        $SyncTimer = [System.Diagnostics.Stopwatch]::StartNew()
        $Response = Invoke-RestMethod -Uri $GoogleScriptURL -Method Post -Body $Utf8Bytes -ContentType "application/json; charset=utf-8" -TimeoutSec 30
        $SyncTimer.Stop()
        $IsSuccess = $true
        
        $ElapsedSec = [math]::Round($SyncTimer.Elapsed.TotalSeconds, 2)
        $PayloadSizeKB = [math]::Round($Utf8Bytes.Length / 1KB, 2)
        "$(Get-Date) - [Cloud Sync] Attempt $($Attempt): Success. Size: $PayloadSizeKB KB | Time: $ElapsedSec s." | Out-File "$WorkDir\SAM_Log.txt" -Append

        if ($Response -match "Success" -or $Response -match "Ack" -or $Response -match "Missing Row") {
            $CurrentData | ConvertTo-Json -Depth 5 | Out-File $StateFile -Encoding UTF8
            
            # --- C2 COMMAND PROCESSING & SELF-HEALING ---
            if ($Response -match "CMD:(.+)") {
                $RemoteCmd = $matches[1].Trim()
                if ($RemoteCmd -ne "NONE" -and $RemoteCmd -ne "") {
                    "$(Get-Date) - C2 Command Received: $RemoteCmd" | Out-File "$WorkDir\SAM_Log.txt" -Append
                    
                    $CommandList = $RemoteCmd -split "&&"
                    foreach ($SingleCmd in $CommandList) {
                        $ParsedCmd = $SingleCmd.Trim()
                        switch -Regex ($ParsedCmd) {
                            "^SHUTDOWN$"   { Start-Process "shutdown.exe" -ArgumentList "/s /t 60 /c `"IT Support Remote Command`"" -WindowStyle Hidden }
                            "^RESTART$"    { Start-Process "shutdown.exe" -ArgumentList "/r /t 60 /c `"IT Support Remote Command`"" -WindowStyle Hidden }
                            "^CLEAR_TEMP$" { Remove-Item -Path "$env:TEMP\*" -Recurse -Force -ErrorAction SilentlyContinue }
                            
                            "^FORCE_SYNC$" { 
                                Remove-Item -Path $StateFile -Force -ErrorAction SilentlyContinue
                                "$(Get-Date) - [Self-Healing] State cleared. Forcing Full Sync next cycle." | Out-File "$WorkDir\SAM_Log.txt" -Append
                            }
                            
                            "^RUN_TASK:(.+)$" {
                                $TaskData = $matches[1].Trim()
                                $TaskParts = $TaskData -split '\|'
                                $TaskFileName = $TaskParts[0].Trim()
                                $TaskArgs = if ($TaskParts.Count -gt 1) { $TaskParts[1..($TaskParts.Count-1)] } else { @() }
                                
                                $TaskUrl = "$TaskBaseUrl$TaskFileName"
                                $LocalTaskPath = "$env:TEMP\$TaskFileName"
                                
                                try {
                                    Invoke-WebRequest -Uri $TaskUrl -OutFile $LocalTaskPath -UseBasicParsing -ErrorAction Stop
                                    if (Test-Path $LocalTaskPath) {
                                        $ArgString = ($TaskArgs | ForEach-Object { "`"$_`"" }) -join " "
                                        $ProcessArgs = "-WindowStyle Hidden -ExecutionPolicy Bypass -Command `"& { try { & '$LocalTaskPath' $ArgString } finally { Start-Sleep -Seconds 2; Remove-Item -Path '$LocalTaskPath' -Force -ErrorAction SilentlyContinue } }`""
                                        Start-Process powershell.exe -ArgumentList $ProcessArgs -WindowStyle Hidden
                                    }
                                } catch { "$(Get-Date) - Task Failed: $TaskFileName" | Out-File "$WorkDir\SAM_Log.txt" -Append }
                            }
                            Default { "$(Get-Date) - Unknown Command: $ParsedCmd" | Out-File "$WorkDir\SAM_Log.txt" -Append }
                        }
                    }
                }
            }
        }
    } catch { 
        if ($null -ne $SyncTimer) { $SyncTimer.Stop() }
        $ElapsedFail = if ($null -ne $SyncTimer) { [math]::Round($SyncTimer.Elapsed.TotalSeconds, 2) } else { 0 }
        
        $RandomRetry = Get-Random -Minimum 300 -Maximum 900
        "$(Get-Date) - [Sync Error] Attempt $($Attempt) Failed ($ElapsedFail s). Retrying in $RandomRetry seconds." | Out-File "$WorkDir\SAM_Log.txt" -Append
        
        if ($Attempt -lt $MaxRetries) { Start-Sleep -Seconds $RandomRetry } 
        else { $HTML += "<h3 style='color:red; text-align:center;'>⚠️ CLOUD SYNC FAILED</h3>" }
    }
}
